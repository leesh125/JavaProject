switch 문


    switch (식) {
        case 값1 :
            식의 결과가 값1에 매치되는 경우 수행할 코드
            break;
        case 값2 :
            식의 결과가 값2에 매치되는 경우 수행할 코드
            break;
        case 값3 :
            식의 결과가 값3에 매치되는 경우 수행할 코드
            break;
        default:
            식의 결과가 어떠한 case에도 매치되지 않는 경우
    }

예제 코드


        switch (x) {
            case 1 :
                document.getElementById("res2").innerHtml = "1 입니다.";
                break;
            case 2 :
                document.getElementById("res2").innerHtml = "2 입니다.";
                break;
            default:
                document.getElementById("res2").innerHtml = "1 또는 2가 아닙니다.";
        }

for ... in 문


        for(변수명 in 객체) {
            객체가 가지는 속성에 대하여 반복을 진행
            반복 수행할 코드
        }

예제 코드


            var obj = {a: 1, b: 2, c: 3, d: 4}
            for(var name in obj) {
                document.getElementById("res5").innerHTML += name + "<br>";
            }

for ... of 문


        for(변수명 of 객체) {
            반복가능한 객체에 담긴 값을 사용하여 진행
            반복 수행할 코드
        }
